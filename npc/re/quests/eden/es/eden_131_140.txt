mes "Pero me odio más a mí mismo porque quiero estar cerca de ella aunque nunca me vea. Jaja, ¿hablaba demasiado en serio?";
				next;
mes "[Ragi]";
mes "Solo soy un contador de historias. Cualquier historia puede ser inventada. La mayor parte del ti contiene la verdad.";
			}
		}
	}
	close2;
	cutin "",255;
	end;
S_Quest:
	if (isbegin_quest(getarg(0)))
		erasequest getarg(0);
	setquest getarg(1);
	if (getarg(2,0))
		getexp getarg(2),getarg(3);
	return;
S_Erasequest:
	.@total = getargcount();
	for ( .@i = 0; .@i < .@total; .@i += 3 ) {
		.@quest_id = getarg(.@i);
		switch( checkquest(.@quest_id,HUNTING) ) {
		case -1:
			if (checkquest(.@quest_id+1) > -1)
				erasequest (.@quest_id+1);
			continue;
		case 0:
		case 1:
mes "[Ragi]";
			switch( getarg(.@i+1) ) {
			case 1:
mes "El resto de la historia de "+ getarg(.@i+2) +" es Está bien, le preguntaré a alguien más.";
				break;
			case 2:
			case 3:
mes "Preguntaré a otros aventureros sobre la historia de "+ getarg(.@i+2) +".";
				break;
			}
			break;
		case 2:
mes "[Ragi]";
			switch( getarg(.@i+1) ) {
			case 1:
mes "¿Escuchaste la historia de "+ getarg(.@i+2) +"? Gracias. Prometo hacer una buena canción.";
				getexp 1100000,300000;
				break;
			case 2:
mes "¿Es la historia de Marruecos lo suficientemente buena para las canciones? Hiciste un buen trabajo.";
				getexp 1100000,200000;
				break;
			case 3:
mes "¿Cómo fue tu aventura? ¿Fue divertido? Por favor, déjame escuchar tu historia de aventuras algún día.";
				getexp 1100000,200000;
				break;
			}
		}
		erasequest .@quest_id;
		next;
	}
	return;
}
